Problem 1: makeBricks


public boolean makeBricks(int small, int big, int goal) {
    if (small + big * 5 < goal)
        return false;
    if (goal % 5 > small)
        return false;
    return true;
}

Problem 2: loneSum


public int loneSum(int a, int b, int c) {
    if (a == b && a == c) {
        return 0;
    }
    if (a == b) {
        return c;
    }
    if (b == c) {
        return a;
    }
    if (c == a) {
        return b;
    }
    return a + b + c;
}

Problem 3: luckySum


public int luckySum(int a, int b, int c) {
    if ((a == 13 && c == 13) || (a == 13 && b == 13)) {
        return 0;
    }
    if (a == 13) {
        return 0;
    }
    if (b == 13) {
        return a;
    }
    if (c == 13) {
        return a + b;
    }
    return a + b + c;
}

Problem 4: noTeenSum


public int noTeenSum(int a, int b, int c) {
    return fixTeen(a) + fixTeen(b) + fixTeen(c);
}

public int fixTeen(int n) {
    if (n >= 13 && n <= 14 || n >= 17 && n <= 19) {
        return 0;
    }
    return n;
}

Problem 5: roundSum


public int roundSum(int a, int b, int c) {
    return round10(a) + round10(b) + round10(c);
}

public int round10(int num) {
    if (num % 10 < 5 || num % 10 == 0) {
        return (num / 10) * 10;
    }
    if (num % 10 >= 5) {
        return ((num / 10) + 1) * 10;
    }
    return num;
}

Problem 6: closeFar


public boolean closeFar(int a, int b, int c) {
    if (Math.abs(b - c) <= 1)
        return false;
    if (Math.abs(a - b) > 1)
        return true;
    if (Math.abs(a - c) > 1)
        return true;
    return false;
}

Problem 7: blackjack


public int blackjack(int a, int b) {
    if (a > 21 && b > 21)
        return 0;
    if (a > 21 || b > 21) {
        return Math.min(a, b);
    }
    return Math.max(a, b);
}

Problem 8: evenlySpaced


public boolean evenlySpaced(int a, int b, int c) {
    int temp;
    if (a > b) {
        temp = a;
        a = b;
        b = temp;
    }
    if (b > c) {
        temp = b;
        b = c;
        c = temp;
    }
    if (a > b) {
        temp = a;
        a = b;
        b = temp;
    }
    return Math.abs(a - b) == Math.abs(b - c);
}

Problem 9: makeChocolate


public int makeChocolate(int small, int big, int goal) {
    if (small + big * 5 < goal)
        return -1;
    if (goal - big * 5 >= 0) {
        int n = goal - big * 5;
        if (n <= small) {
            return n;
        }
    } else {
        int n = goal % 5;
        if (n <= small) {
            return n;
        }
    }
    return -1;
}
